	script.Parent.TextBox:CaptureFocus()
	
	
	local TextBox = script.Parent.TextBox
	
	local svxs = 1
	
	
	local uis = game:GetService("UserInputService")
	
	uis.InputBegan:Connect(function(Input, gameProcessedEvent)
		if gameProcessedEvent then return end
		if Input.KeyCode == Enum.KeyCode.Q and script.Parent.Visible == true then
			script.Disabled = true
	
			script.Parent:TweenPosition(
				UDim2.new(-0.192, 0, 0.913, 0),
				Enum.EasingDirection.InOut,
				Enum.EasingStyle.Quad,
				0.3,
				true
			)
			wait(0.5)
			script.Parent.Visible = false
			script.Disabled = false	
			wait(0.3)
			TextBox.Text = " "
	
		else
			if Input.KeyCode == Enum.KeyCode.Q and script.Parent.Visible == false then
				script.Disabled = true
				script.Parent.Visible = true
	
				script.Parent:TweenPosition(
					UDim2.new(0.001, 0, 0.913, 0),
					Enum.EasingDirection.InOut,
					Enum.EasingStyle.Quad,
					0.3,
					true
				)
				wait(0.5)
	
				script.Parent.TextBox:CaptureFocus()
	
				script.Disabled = false
				script.Parent.Visible = true
			end
		end
	end)
	
	uis.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.Keyboard then
			if input.KeyCode == Enum.KeyCode.Return then
				script.Disabled = true
	
				script.Parent:TweenPosition(
					UDim2.new(-0.192, 0, 0.913, 0),

			end
		end
	end)

		script.Disabled = true
	
		script.Parent:TweenPosition(
			UDim2.new(-0.192, 0, 0.913, 0),
			Enum.EasingDirection.InOut,
			Enum.EasingStyle.Quad,
			0.3,
			true
		)
		wait(0.5)
		script.Parent.Visible = false
		script.Disabled = false	
		wait(0.3)

function healthBar()
	local StarterGui = game:GetService("StarterGui")

	StarterGui:SetCoreGuiEnabled(Enum.CoreGuiType.Health, false)
	local Player = game.Players.LocalPlayer 
	local Character = Player.Character
	local Humanoid = Character:FindFirstChild("Humanoid")
	
	local Bar = Bar
	
	
	local uwuHealth = math.floor(Humanoid.Health/Humanoid.MaxHealth * 100)
	
	if uwuHealth == 100 then
		Holder.Visible = false
	end
	repeat wait() until Player.Character or Player.CharacterAdded:Wait()

	

	local GreenColor = Color3.fromRGB(0, 255, 0)
	local RedColor = Color3.fromRGB(255, 0, 0)
	local OrangeColor = Color3.fromRGB(255, 170, 0)
	local YellowColor = Color3.fromRGB(255, 255, 0)

	local cHealth = Humanoid.Health

	Humanoid.HealthChanged:Connect(function(Health)

		local Health = math.floor(Health)
		Health = math.floor(Humanoid.Health/Humanoid.MaxHealth * 100)
		
		Bar.Size = UDim2.new(0, Health, 0, 3)

		prcvida.Text = tostring(Health .. "%")
		
		if Health == 100 then
			Holder.Visible = false
		end
		
		if Health < 100 then
			Holder.Visible = true
		end
		
		if Health > 75 then
			Bar.BackgroundColor3 = GreenColor

		elseif Health < 74 and Health > 40 then
			Bar.BackgroundColor3 = YellowColor
		elseif Health < 40 and Health > 20 then
			Bar.BackgroundColor3 = OrangeColor

		elseif Health < 20 then
			Bar.BackgroundColor3 = RedColor
		end

		if Health < cHealth then
			local dmg = Dmg

			dmg.BackgroundTransparency = 0.9
			wait(.06)
			dmg.BackgroundTransparency = 1
			cHealth = Health

			if states.DAMAGESOUNDS == true then
				local hitsound = math.random(1, 5)
				if hitsound == 1 then
					createSoundEffect("rbxassetid://6978732312", 1)
				else
					if hitsound == 2 then
						createSoundEffect("rbxassetid://6978732129", 1)
					else
						if hitsound == 3 then
							createSoundEffect("rbxassetid://6978731977", 1)
						else
							if hitsound == 4 then
								createSoundEffect("rbxassetid://6978731806", 1)
							else
								if hitsound == 5 then
									createSoundEffect("rbxassetid://6978732459", 1)
								end
							end
						end
					end
				end
			end
			
		end
	end)
end

local function GetPlayerFromString(nameString)
	if not nameString then return end 
	local matches= {}
	for _, player in ipairs (game.Players:GetPlayers()) do
		if string.lower(player.Name):match(string.lower(nameString)) then
			table.insert(matches, player)
		end
	end
	if #matches== 1 then
		return matches[1]
	else
		return nil
	end
end

local function GetNameFromList(nameString, list)
	if not nameString then return end 
	local matches= {}
	for _, jeje in pairs (list) do
		if string.lower(jeje):match(string.lower(nameString)) then
			table.insert(matches, jeje)
		end
	end
	if #matches== 1 then
		return matches[1]
	else
		return nil
	end
end

	local ShotHit = false

		local Hit = bullet.Hit

		local Limb = Hit.Parent:FindFirstChildOfClass('Humanoid') ~= nil
		local Accessory = Hit.Parent.Parent:FindFirstChildOfClass('Humanoid') ~= nil

		if Limb then
			local Player = Players:GetPlayerFromCharacter(Hit.Parent)

			if Player.TeamColor ~= LocalPlayer.TeamColor then
				ShotHit = true
				break
			end
		elseif Accessory then
			local Player = Players:GetPlayerFromCharacter(Hit.Parent.Parent)

			if Player.TeamColor ~= LocalPlayer.TeamColor then
				ShotHit = true
				break
			end
		end
	end

		if states.HITSOUND == true then
			createSoundEffect("rbxassetid://"..idHitSound, 1)

			local Clone = Hitmarker:Clone()
			Clone.Position = UDim2.new(0,Mouse.X,0,Mouse.Y)

			wait()
			Clone:Destroy()
		end
	end
end)
